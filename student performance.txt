import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

data = pd.read_csv("StudentsPerformance.csv")

print(data.head())


print("\nMissing values:\n", data.isnull().sum())


print("\nData Summary:\n", data.describe())

plt.figure(figsize=(8,5))
sns.histplot(data['math score'], bins=20, kde=True, color='skyblue')
plt.title("Distribution of Math Scores")
plt.show()

sns.scatterplot(x='reading score', y='writing score', data=data, color='orange')
plt.title("Reading vs Writing Score")
plt.show()

avg_by_gender = data.groupby('gender')[['math score', 'reading score', 'writing score']].mean()
print("\nAverage scores by gender:\n", avg_by_gender)

avg_by_gender.plot(kind='bar', figsize=(8,5))
plt.title("Average Scores by Gender")
plt.ylabel("Score")
plt.show()

plt.figure(figsize=(10,5))
sns.boxplot(x='parental level of education', y='math score', data=data, palette='Set3')
plt.xticks(rotation=45)
plt.title("Parental Education vs Math Score")
plt.show()

plt.figure(figsize=(6,4))
sns.heatmap(data[['math score', 'reading score', 'writing score']].corr(), annot=True, cmap='coolwarm')
plt.title("Correlation Between Scores")
plt.show()
